OpenSSL使用笔记-20211226

OpenSSL使用笔记

openssl ecparam -name prime256v1 -genkey -noout -out client.key  #生成私钥和公钥
openssl ec -text -noout -in client.key  #查看生成的私钥和公钥
openssl req -new -newkey -x509 -sha256 -key client.key -out client.csr -subj '/C=AU/ST=Some-State/L=/O=cert_center/OU=/CN=selfsigncsr/emailAddress='  #生成csr
openssl req -noout -text -in client.csr  #查看csr
openssl req -new -x509 -sha256 -key ca.key -out ca.cer -days 10000 -subj '/C=AU/ST=Some-State/L=/O=cert_center/OU=/CN=selfsignca/emailAddress='  #生成自签名CA证书
openssl x509 -in ca.cer -text -noout  #查看自签名证书

mkdir -p ./demoCA/newcerts
touch ./demoCA/index.txt
> ./demoCA/serial echo 00
openssl ca -cert ca.cer -keyfile ca.key -in client.csr -out client.cer -days 3650  #对csr签名

cat client.key >> client.cer
openssl pkcs12 -export -in client.cer -out client.p12 #转p12格式
